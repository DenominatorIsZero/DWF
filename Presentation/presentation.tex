\section{Grundlagen}

\begin{frame}
\frametitle{Bireziproker Teilbandfilter}
\begin{itemize}
	\item Unterklasse der Wellendigitalfilter
	\item Durchlassdämpfung oder Sperrdämpfung frei wählbar
	\item N - 1 / 2 Adaptoren
	\item Verwendung von Cauer-Parametern
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Oktav-Filterbank}
\begin{itemize}
	\item Besteht aus mehreren Brücken-Wellendigitalfiltern
	\item Halbierung des Frequenzbandes pro Teilfilterband
	\item für jedes Teilband Filtertransformation
	\item Daraus folgt: Für jede Stufe Verdopplung der Delays
	\item Verzögerungen nach den Teilbändern, um die Phasenverschiebung auszugleichen
\end{itemize}
\end{frame}

\subsection{Optimierung}
    \begin{frame}
        \begin{itemize}
            \item Erster Ansatz: ohne Optimierung
            \begin{itemize}
                \item DMA Blockgröße: 8 Samples
                \item Delaylines als lineare Puffer realisiert
                \item $\rightarrow$ 0 Teilbandfilter
            \end{itemize}
        \end{itemize}
        \begin{itemize}
            \item 1. Optimierungsschritt
            \begin{itemize}
                \item Compileroptimierung auf Stufe 1
                \item $\rightarrow$ 3 Teilbandfilter
            \end{itemize}
        \end{itemize}
        \begin{itemize}
            \item 2. Optimierungsschritt
            \begin{itemize}
                \item Delaylines für Adaptore als Ringpuffer realisiert
                \item $\rightarrow$ 6 Teilbandfilter
            \end{itemize}
        \end{itemize}
    \end{frame}

    \begin{frame}
        \begin{figure}[p]
            \centering
            \includegraphics[width=0.7\textwidth]{img/ringbuffer.png}
            \caption{Schematische Darstellung eines Ringpuffers}
        \end{figure}
    \end{frame}
    
    \begin{frame}
        \begin{figure}[p]
            \centering
            \includegraphics[width=1.1\textwidth]{img/entzerrer.png}
            \caption{Entzerrer mit einfacher Delayline}
        \end{figure}
    \end{frame}

    \begin{frame}
        \begin{itemize}
            \item Entzerrung hinzugefügt
            \begin{itemize}
                \item DMA Blockgröße: 2048 Samples
                \item $\rightarrow$ 6 Teilbandfilter
            \end{itemize}
            \item Letzter Optimierungsschritt
            \begin{itemize}
                \item Sämtliche Delaylines als Ringpuffer realisiert
                \item Ersetzen der Modulo-Funktion
                \item $\rightarrow$ 7 Teilbandfilter
            \end{itemize}
        \end{itemize}
        \begin{itemize}
            \item Weitere Optimierungsmaßnahmen ohne Wirkung
            \begin{itemize}
                \item Vergößerung der Ein- und Ausgabepuffer
                \item Hinzufügen eines dritten Speicherblocks
                \item Verringerung der Samplerate
            \end{itemize}
            \item Rechenzeit reicht für ein 8. Teilbandfilter ``gerade so'' nicht aus.
        \end{itemize}
    \end{frame}
